# Core MCP Servers for Trading Bot

# Memory Server - Essential for trade learning
claude mcp add memory "npx" "@modelcontextprotocol/server-memory"

# SQLite Database - Primary data storage (using D: drive)
claude mcp add sqlite "npx" "@modelcontextprotocol/server-sqlite" "D:/trading_data/trading_bot.db"

# Time/Scheduler Server - For scheduled operations
claude mcp add time "npx" "@modelcontextprotocol/server-time"

# File System Access - For bot code and data access
claude mcp add filesystem "npx" "@modelcontextprotocol/server-filesystem" "/mnt/c/dev/tools/crypto-trading-bot-2025"

# Trading Data Access - For D: drive data
claude mcp add filesystem-data "npx" "@modelcontextprotocol/server-filesystem" "D:/trading_data"

# Custom Trading MCP Server (our extension)
claude mcp add trading-mcp "python" "extensions/mcp-trader/src/mcp_trader/server.py"

# CCXT MCP Server (for additional exchange support)
claude mcp add ccxt-mcp "node" "extensions/ccxt-mcp/src/index.ts"

# Verification Commands
claude mcp list
claude mcp test

# Start MCP server session
claude mcp serve
